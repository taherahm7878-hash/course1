الجمل الشرطية

أمثله

وإن لم يتحقق كلا الشرطين فنضيف شرطا أخيراً بدون شرط

المثال الأول

let age = 5;
if (age > 15) {
  console.log("Adult");
} else if (age > 30) {
  console.log("Man");
} else {
  console.log("Un Known");
}

بما أن العمر أكبر من 15 فيطبع الأمر الذي داخل الكونسول

النتيجة 
        Un Known

المثال الثاني

let a = 10;
a = a + 10;
b = 10;
if (b + a == 3 * 11) {
  console.log("A");
} else if (b + 10 == a + 1) {
  console.log("B");
} else {
  console.log("C");
}
النتيجة 
            C

المثال الثالث

let age = 17;
age = 18;
if (age == 10) {
  console.log("Baby");
} else if (age >= 18) {
  console.log("Adult")
} else {
  console.log("R");
}
النتيجة 
       Adult
