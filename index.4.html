الجمل الشرطية

نستخدم الجمل الشرطية للتحقق

أمثله

المثال الأول

let age = 18;
if (age > 15) {
  console.log("Adult");
}

بما أن العمر أكبر من 15 فيطبع الأمر الذي داخل الكونسول

النتيجة 
        Adult
OR

المثال الثاني

let a = 10;
a = a + 10;
b = 10;
if (b + a == 3 * 10) {
  console.log("Good");
}
النتيجة 
            Good

اذا لم يتحقق الشرط الأول فندخل على الشرط الثاني

المثال الثالث

let age = 17;
age = 18;
if (age == 10) {
  console.log("Baby");
} else if (age >= 18) {
  console.log("Adult")
}
النتيجة 
       Adult
